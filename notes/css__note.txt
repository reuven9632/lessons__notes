СПОСОБЫ ПОЗИЦИОНИРОВАНИЯ ЭЛЕМЕНТОВ В БЛОКЕ header


Способ #1:
.header__inner{
	display: flex;                       	//устанавливает новые правила для него и всех дочерних элементов
	justify-content: space-between;		//устанавливает расстояние между его дочерними блоками
}
.header__inner-text-center{			//дочерний блок стоящий по центру(как его примкнуть к лквому элементу)
	margin-right: auto;			//отталкивает элемет справа к левому краю максимально
	margin-left: 30px;			//отталкивает элемет слева на  30px для растояния между элементами
}


Способ #2:
.header__inner{
	overflow: hidden;			//обьявляя	overflow: hidden;	родителю мы предотвращаем проваливание элементов
}
.header__logo{
	float: left;                       	//устанавливает прижатие элемента к левому краю
}
.header__text-center{				//дочерний блок стоящий по центру(как его примкнуть к лквому элементу)
	float: left;				//устанавливает прижатие элемента к левому краю
}
.header__mail{
	float: right;				//устанавливает прижатие элемента к правому краю
}


Способ #3:
.header__inner{
	float: left;   			//обьявляя	float: left;	родителю мы предотвращаем проваливание элементов, но нужно последующему блоку задать	clear: both;
}
.header__logo{
	float: left;                       	//устанавливает прижатие элемента к левому краю
}
.header__text-center{				//дочерний блок стоящий по центру(как его примкнуть к лквому элементу)
	float: left;				//устанавливает прижатие элемента к левому краю
}
.header__mail{
	float: right;				//устанавливает прижатие элемента к правому краю
}
.section__next{
	clear: both;				//отменяет обтекание float-ов с обоих сторон
}
